# To check temp path
for i in /sys/class/hwmon/hwmon*/temp*_input
    set dir (dirname $i)
    set name (cat $dir/name)

    # Remove the '_input' suffix to get the prefix
    set prefix (string replace '_input' '' $i)
    set label_file $prefix'_label'

    if test -e $label_file
        set label (cat $label_file 2>/dev/null)
    else
        set base_name (basename $prefix)
        set label $base_name
    end

    set full_path (readlink -f $i)
    echo "$name: $label $full_path"
end
